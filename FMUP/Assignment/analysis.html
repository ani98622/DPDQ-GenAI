# 1. Chi-square Tests
# ---------------------------

def run_chi_square(col1, col2):
    contingency = pd.crosstab(df[col1], df[col2])
    chi2, p, dof, expected = chi2_contingency(contingency)
    print(f"\nChi-square Test: {col1} vs {col2}")
    print("Chi2 Value:", round(chi2, 2))
    print("Degrees of Freedom:", dof)
    print("P-value:", round(p, 4))
    if p < 0.05:
        print("=> Significant relationship (Reject null hypothesis)")
    else:
        print("=> No significant relationship (Fail to reject null hypothesis)")

# Productivity relations
run_chi_square("remote_work_duration", "productivity_rating")
run_chi_square("manager_feedback", "productivity_rating")
run_chi_square("tools_resources", "productivity_rating")
run_chi_square("difficulties_frequency", "productivity_rating")

# Engagement relations
run_chi_square("remote_work_duration", "engagement_with_goals")
run_chi_square("team_collaboration", "engagement_with_goals")
run_chi_square("manager_feedback", "engagement_with_goals")
